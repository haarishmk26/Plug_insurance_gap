# Phase 1 MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking. Read `PRD.md` and `memory.md` before starting any task.

**Goal:** Build Phase 1 of the District Cover Insurance Readiness Platform — a Telegram bot that collects ACORD 125 intake data from SF business owners, paired with a broker web dashboard for review and PDF export.

**Architecture:** Next.js App Router (broker dashboard + API routes) backed entirely by Supabase (Postgres, Auth, Storage). The Telegram bot runs as a Next.js API route receiving webhooks; for local dev, ngrok exposes it. The score engine is pure TypeScript logic called server-side. PDF export uses `@react-pdf/renderer` in a Next.js API route.

**Tech Stack:** Next.js 14 (App Router), TypeScript, Shadcn/UI, Tailwind CSS, Supabase (Postgres + Auth + Storage), Grammy (Telegram bot framework), @react-pdf/renderer, ngrok (local dev only)

---

## Phase 1 Scope (from PRD Section 11)

In scope:
- Telegram bot: deep link initiation, conversational ACORD A–E intake, photo/PDF upload, resumable sessions
- Broker dashboard: login, client list, new client form + deep link generation, client detail view (dossier)
- Score engine: 5-dimension 0–100 score from intake data
- PDF export: ACORD fields + uploads list + score
- Broker notification when intake completes

Out of scope (Phase 2+): advisory Q&A, score/checklist delivery to bot, SF public data, AI photo verification, Kanban pipeline, Insly integration.

```sql
-- Brokers are Supabase Auth users with a profile
create table public.brokers (
  id uuid primary key references auth.users(id) on delete cascade,
  name text not null,
  email text not null,
  created_at timestamptz default now()
);

-- RLS: brokers can only read/write their own row
alter table public.brokers enable row level security;
create policy "brokers_own_row" on public.brokers
  using (auth.uid() = id);

-- Clients (business owners) belong to a broker
create table public.clients (
  id uuid primary key default gen_random_uuid(),
  broker_id uuid references public.brokers(id) on delete cascade not null,
  business_name text not null,
  owner_name text not null,
  owner_contact text,
  telegram_chat_id bigint,
  intake_token text unique not null,
  intake_status text not null default 'pending'
    check (intake_status in ('pending', 'in_progress', 'complete')),
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

alter table public.clients enable row level security;
create policy "brokers_own_clients" on public.clients
  using (broker_id = auth.uid());

-- ACORD 125 intake answers stored per section
create table public.intake_data (
  id uuid primary key default gen_random_uuid(),
  client_id uuid references public.clients(id) on delete cascade unique not null,
  section_a jsonb not null default '{}',
  section_b jsonb not null default '{}',
  section_c jsonb not null default '{}',
  section_d jsonb not null default '{}',
  section_e jsonb not null default '{}',
  current_question_key text,
  updated_at timestamptz default now()
);

alter table public.intake_data enable row level security;
create policy "brokers_own_intake" on public.intake_data
  using (
    exists (
      select 1 from public.clients c
      where c.id = client_id and c.broker_id = auth.uid()
    )
  );

-- Uploaded files (photos + PDFs)
create table public.uploads (
  id uuid primary key default gen_random_uuid(),
  client_id uuid references public.clients(id) on delete cascade not null,
  storage_path text not null,
  file_name text not null,
  file_type text not null check (file_type in ('photo', 'pdf')),
  label text,
  telegram_file_id text,
  uploaded_at timestamptz default now()
);

alter table public.uploads enable row level security;
create policy "brokers_own_uploads" on public.uploads
  using (
    exists (
      select 1 from public.clients c
      where c.id = client_id and c.broker_id = auth.uid()
    )
  );

-- Readiness scores (one per client, recalculated on demand)
create table public.readiness_scores (
  id uuid primary key default gen_random_uuid(),
  client_id uuid references public.clients(id) on delete cascade unique not null,
  total_score integer not null check (total_score between 0 and 100),
  documentation_score integer not null check (documentation_score between 0 and 25),
  safety_score integer not null check (safety_score between 0 and 25),
  property_score integer not null check (property_score between 0 and 20),
  claims_score integer not null check (claims_score between 0 and 20),
  neighborhood_score integer not null check (neighborhood_score between 0 and 10),
  score_details jsonb not null default '{}',
  calculated_at timestamptz default now()
);

alter table public.readiness_scores enable row level security;
create policy "brokers_own_scores" on public.readiness_scores
  using (
    exists (
      select 1 from public.clients c
      where c.id = client_id and c.broker_id = auth.uid()
    )
  );

-- Trigger: update clients.updated_at on row change
create or replace function public.set_updated_at()
returns trigger as $$
begin new.updated_at = now(); return new; end;
$$ language plpgsql;

create trigger clients_updated_at before update on public.clients
  for each row execute function public.set_updated_at();
create trigger intake_updated_at before update on public.intake_data
  for each row execute function public.set_updated_at();
```

import { createServerClient } from '@supabase/ssr'
import { cookies } from 'next/headers'

export async function createClient() {
  const cookieStore = await cookies()
  return createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() { return cookieStore.getAll() },
        setAll(cookiesToSet) {
          try {
            cookiesToSet.forEach(({ name, value, options }) =>
              cookieStore.set(name, value, options)
            )
          } catch {}
        },
      },
    }
  )
}

- [ ] **Step 3: Create `middleware.ts` — protect dashboard routes**

```typescript
import { createServerClient } from '@supabase/ssr'
import { NextResponse, type NextRequest } from 'next/server'

export async function middleware(request: NextRequest) {
  let supabaseResponse = NextResponse.next({ request })

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() { return request.cookies.getAll() },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value }) => request.cookies.set(name, value))
          supabaseResponse = NextResponse.next({ request })
          cookiesToSet.forEach(({ name, value, options }) =>
            supabaseResponse.cookies.set(name, value, options)
          )
        },
      },
    }
  )

  const { data: { user } } = await supabase.auth.getUser()

  if (!user && !request.nextUrl.pathname.startsWith('/login')) {
    return NextResponse.redirect(new URL('/login', request.url))
  }

  return supabaseResponse
}

export const config = {
  matcher: ['/((?!_next/static|_next/image|favicon.ico|api/).*)'],
}

'use client'

import { useState } from 'react'
import { useRouter } from 'next/navigation'
import { createClient } from '@/lib/supabase/client'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { Label } from '@/components/ui/label'
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'
import { toast } from 'sonner'

export default function LoginPage() {
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [loading, setLoading] = useState(false)
  const router = useRouter()
  const supabase = createClient()

  async function handleLogin(e: React.FormEvent) {
    e.preventDefault()
    setLoading(true)
    const { error } = await supabase.auth.signInWithPassword({ email, password })
    if (error) {
      toast.error(error.message)
    } else {
      router.push('/')
      router.refresh()
    }
    setLoading(false)
  }

  return (
    <Card className="w-full max-w-sm">
      <CardHeader>
        <CardTitle className="text-2xl">District Cover</CardTitle>
        <p className="text-sm text-muted-foreground">Broker portal sign in</p>
      </CardHeader>
      <CardContent>
        <form onSubmit={handleLogin} className="space-y-4">
          <div className="space-y-1">
            <Label htmlFor="email">Email</Label>
            <Input id="email" type="email" value={email}
              onChange={e => setEmail(e.target.value)} required />
          </div>
          <div className="space-y-1">
            <Label htmlFor="password">Password</Label>
            <Input id="password" type="password" value={password}
              onChange={e => setPassword(e.target.value)} required />
          </div>
          <Button type="submit" className="w-full" disabled={loading}>
            {loading ? 'Signing in...' : 'Sign in'}
          </Button>
        </form>
      </CardContent>
    </Card>
  )
}
```

// Returns the next question key given the current one, or null if intake is complete
export function getNextQuestionKey(
  currentKey: string | null,
  answers: Record<string, unknown>
): string | null {
  if (!currentKey) return ALL_QUESTIONS[0].key

  const currentIndex = ALL_QUESTIONS.findIndex(q => q.key === currentKey)
  if (currentIndex === -1) return null

  for (let i = currentIndex + 1; i < ALL_QUESTIONS.length; i++) {
    const next = ALL_QUESTIONS[i]

    // Skip landlord question if owner (not tenant)
    if (next.key === 'b_landlord') {
      const ownership = answers['b_ownership'] as string
      if (!ownership?.toLowerCase().includes('tenant')) continue
    }

    // Skip sprinkler certificate if no sprinklers
    if (next.key === 'd_sprinkler_certificate') {
      if (answers['d_sprinklers'] !== 'yes') continue
    }

    // Skip alarm contract if no alarm
    if (next.key === 'd_alarm_contract') {
      if (answers['d_alarm_system'] !== 'yes') continue
    }

    // Skip extinguisher photo if no extinguishers
    if (next.key === 'd_extinguisher_photo') {
      if (answers['d_fire_extinguishers'] !== 'yes') continue
    }

    // Skip claims detail if no claims
    if (next.key === 'e_claims_detail') {
      if (answers['e_claims'] !== 'yes') continue
    }

    return next.key
  }

  return null // intake complete
}

import { createServiceClient } from '@/lib/supabase/server'

// Flatten all section answers into a single record for skip logic
export function flattenAnswers(intakeData: {
  section_a: Record<string, unknown>
  section_b: Record<string, unknown>
  section_c: Record<string, unknown>
  section_d: Record<string, unknown>
  section_e: Record<string, unknown>
}): Record<string, unknown> {
  return {
    ...intakeData.section_a,
    ...intakeData.section_b,
    ...intakeData.section_c,
    ...intakeData.section_d,
    ...intakeData.section_e,
  }
}

// Map a question key's section prefix to the DB column
function sectionColumn(key: string): string {
  const prefix = key.split('_')[0]
  const map: Record<string, string> = {
    a: 'section_a', b: 'section_b', c: 'section_c',
    d: 'section_d', e: 'section_e',
  }
  return map[prefix] ?? 'section_a'
}

// Save one answer to the correct section JSONB column
export async function saveAnswer(
  clientId: string,
  questionKey: string,
  answer: string,
  nextQuestionKey: string | null
) {
  const supabase = createServiceClient()
  const col = sectionColumn(questionKey)

  // Fetch current section data
  const { data: existing } = await supabase
    .from('intake_data')
    .select(col)
    .eq('client_id', clientId)
    .single()

  const currentSection = (existing?.[col as keyof typeof existing] ?? {}) as Record<string, unknown>

  await supabase
    .from('intake_data')
    .upsert({
      client_id: clientId,
      [col]: { ...currentSection, [questionKey]: answer },
      current_question_key: nextQuestionKey,
    })
    .eq('client_id', clientId)
}

// Get a client by their intake token
export async function getClientByToken(token: string) {
  const supabase = createServiceClient()
  const { data } = await supabase
    .from('clients')
    .select('*, intake_data(*)')
    .eq('intake_token', token)
    .single()
  return data
}

// Link telegram_chat_id to client on first message
export async function linkChatId(clientId: string, chatId: number) {
  const supabase = createServiceClient()
  await supabase
    .from('clients')
    .update({ telegram_chat_id: chatId, intake_status: 'in_progress' })
    .eq('id', clientId)
}

// Mark intake as complete and notify broker
export async function markIntakeComplete(clientId: string) {
  const supabase = createServiceClient()

  await supabase
    .from('clients')
    .update({ intake_status: 'complete' })
    .eq('id', clientId)

  // Trigger score calculation (fire and forget)
  fetch(`${process.env.NEXT_PUBLIC_APP_URL}/api/score/${clientId}`, {
    method: 'POST',
    headers: { 'x-internal-secret': process.env.TELEGRAM_WEBHOOK_SECRET! },
  }).catch(() => {}) // non-blocking
}

import { Bot, webhookCallback } from 'grammy'
import { createServiceClient } from '@/lib/supabase/server'
import {
  getClientByToken, linkChatId, saveAnswer,
  markIntakeComplete, flattenAnswers
} from './session'
import { ALL_QUESTIONS, QUESTION_BY_KEY, getNextQuestionKey } from './questions'

const bot = new Bot(process.env.TELEGRAM_BOT_TOKEN!)

// /start command — handles deep link token
bot.command('start', async (ctx) => {
  const token = ctx.match
  if (!token) {
    await ctx.reply("Welcome! This bot is invite-only. Your broker will send you a link to get started.")
    return
  }

  const client = await getClientByToken(token)
  if (!client) {
    await ctx.reply("That link isn't valid or has expired. Please ask your broker for a new one.")
    return
  }

  await linkChatId(client.id, ctx.chat.id)

  // Ensure intake_data row exists
  const supabase = createServiceClient()
  await supabase.from('intake_data').upsert({ client_id: client.id })

  // Greet by name and explain
  await ctx.reply(
    `Hi ${client.owner_name}! 👋\n\n` +
    `I'm here to help ${client.business_name} get ready for commercial insurance.\n\n` +
    `I'll ask you about 30 short questions — one at a time. ` +
    `You can stop and come back anytime; I'll save your progress.\n\n` +
    `Your answers help your broker submit a complete application that tells your full story — ` +
    `not just your ZIP code.\n\nLet's get started!`
  )

  // Send first question
  const firstKey = getNextQuestionKey(null, {})
  if (firstKey) await sendQuestion(ctx, QUESTION_BY_KEY[firstKey])
})

// Handle regular messages (answers to questions)
bot.on('message', async (ctx) => {
  const chatId = ctx.chat.id

  // Find client by chat_id
  const supabase = createServiceClient()
  const { data: client } = await supabase
    .from('clients')
    .select('*, intake_data(*)')
    .eq('telegram_chat_id', chatId)
    .single()

  if (!client) {
    await ctx.reply("Please use your broker's invite link to get started.")
    return
  }

  if (client.intake_status === 'complete') {
    await ctx.reply("Your intake is already complete! Your broker will be in touch soon.")
    return
  }

  const intakeData = client.intake_data?.[0]
  const currentKey = intakeData?.current_question_key

  if (!currentKey) {
    // No current question — start from beginning
    const firstKey = getNextQuestionKey(null, {})
    if (firstKey) await sendQuestion(ctx, QUESTION_BY_KEY[firstKey])
    return
  }

  const currentQuestion = QUESTION_BY_KEY[currentKey]
  if (!currentQuestion) return

  // Handle file uploads
  if (currentQuestion.type === 'upload_photo' || currentQuestion.type === 'upload_pdf' || currentQuestion.type === 'upload_either') {
    const fileId = ctx.message.photo
      ? ctx.message.photo[ctx.message.photo.length - 1].file_id
      : ctx.message.document?.file_id

    if (!fileId) {
      // User skipped upload with text
      const answers = flattenAnswers(intakeData ?? { section_a: {}, section_b: {}, section_c: {}, section_d: {}, section_e: {} })
      const nextKey = getNextQuestionKey(currentKey, answers)

      if (nextKey === null) {
        await handleIntakeComplete(ctx, client.id, client.owner_name)
      } else {
        await saveAnswer(client.id, currentKey, 'skipped', nextKey)
        await sendQuestion(ctx, QUESTION_BY_KEY[nextKey])
      }
      return
    }

    // Download and store file
    const file = await ctx.api.getFile(fileId)
    const fileUrl = `https://api.telegram.org/file/bot${process.env.TELEGRAM_BOT_TOKEN}/${file.file_path}`
    const response = await fetch(fileUrl)
    const buffer = await response.arrayBuffer()

    const ext = file.file_path?.split('.').pop() ?? 'bin'
    const storagePath = `${client.id}/${currentKey}.${ext}`
    const fileType = ctx.message.photo ? 'photo' : 'pdf'

    await supabase.storage
      .from('client-uploads')
      .upload(storagePath, buffer, { upsert: true })

    await supabase.from('uploads').insert({
      client_id: client.id,
      storage_path: storagePath,
      file_name: `${currentKey}.${ext}`,
      file_type: fileType,
      label: currentKey,
      telegram_file_id: fileId,
    })

    await ctx.reply("Got it — saved! ✅")

    const answers = flattenAnswers(intakeData ?? { section_a: {}, section_b: {}, section_c: {}, section_d: {}, section_e: {} })
    const nextKey = getNextQuestionKey(currentKey, answers)

    if (nextKey === null) {
      await handleIntakeComplete(ctx, client.id, client.owner_name)
    } else {
      await saveAnswer(client.id, currentKey, storagePath, nextKey)
      await sendQuestion(ctx, QUESTION_BY_KEY[nextKey])
    }
    return
  }

  // Text answer
  const answer = ctx.message.text ?? ''
  const normalizedAnswer = normalizeAnswer(currentQuestion, answer)

  const answers = {
    ...flattenAnswers(intakeData ?? { section_a: {}, section_b: {}, section_c: {}, section_d: {}, section_e: {} }),
    [currentKey]: normalizedAnswer,
  }
  const nextKey = getNextQuestionKey(currentKey, answers)

  if (nextKey === null) {
    await saveAnswer(client.id, currentKey, normalizedAnswer, null)
    await handleIntakeComplete(ctx, client.id, client.owner_name)
  } else {
    await saveAnswer(client.id, currentKey, normalizedAnswer, nextKey)
    await sendQuestion(ctx, QUESTION_BY_KEY[nextKey])
  }
})

async function sendQuestion(ctx: any, question: any) {
  let text = question.text
  if (question.followUp) text += `\n\n_${question.followUp}_`

  if (question.type === 'yes_no') {
    await ctx.reply(text, {
      reply_markup: {
        keyboard: [[{ text: 'Yes' }, { text: 'No' }]],
        resize_keyboard: true,
        one_time_keyboard: true,
      },
      parse_mode: 'Markdown',
    })
  } else if (question.type === 'choice' && question.choices) {
    const keyboard = question.choices.map((c: string) => [{ text: c }])
    await ctx.reply(text, {
      reply_markup: { keyboard, resize_keyboard: true, one_time_keyboard: true },
      parse_mode: 'Markdown',
    })
  } else if (question.type === 'upload_photo' || question.type === 'upload_pdf' || question.type === 'upload_either') {
    await ctx.reply(text + '\n\n_(Type "skip" if you don\'t have this right now)_', {
      reply_markup: { remove_keyboard: true },
      parse_mode: 'Markdown',
    })
  } else {
    await ctx.reply(text, {
      reply_markup: { remove_keyboard: true },
      parse_mode: 'Markdown',
    })
  }
}

function normalizeAnswer(question: any, answer: string): string {
  if (question.type === 'yes_no') {
    const lower = answer.toLowerCase()
    if (['yes', 'y', '1', 'true'].includes(lower)) return 'yes'
    if (['no', 'n', '0', 'false'].includes(lower)) return 'no'
  }
  return answer.trim()
}

async function handleIntakeComplete(ctx: any, clientId: string, ownerName: string) {
  await ctx.reply(
    `That's everything, ${ownerName}! 🎉\n\n` +
    `Your intake is complete. Your broker has been notified and will review your profile shortly.\n\n` +
    `If you have any questions in the meantime, feel free to message here.`
  )
  await markIntakeComplete(clientId)
}

import type { IntakeData } from '@/lib/supabase/types'

export interface ScoreResult {
  total_score: number
  documentation_score: number  // 0-25
  safety_score: number         // 0-25
  property_score: number       // 0-20
  claims_score: number         // 0-20
  neighborhood_score: number   // 0-10 (Phase 1: fixed at 5 — no SF data yet)
  score_details: Record<string, { points: number; max: number; explanation: string }>
}

function answers(data: IntakeData): Record<string, unknown> {
  return {
    ...data.section_a,
    ...data.section_b,
    ...data.section_c,
    ...data.section_d,
    ...data.section_e,
  }
}

// Documentation Completeness — 25 pts
// Source: Incomplete ACORD submissions are #1 cause of underwriting delays.
// District Cover framing: businesses are misunderstood, not too risky.
function scoreDocumentation(a: Record<string, unknown>): { score: number; detail: Record<string, unknown> } {
  const required = [
    'a_business_name', 'a_entity_type', 'a_fein', 'a_date_started', 'a_phone',
    'b_address', 'b_ownership', 'b_total_sqft', 'b_occupied_sqft', 'b_public_sqft',
    'b_annual_revenue', 'b_fulltime_employees', 'b_operations',
    'e_prior_carrier', 'e_prior_policy_number', 'e_prior_policy_dates', 'e_prior_premium',
  ]
  const filled = required.filter(k => a[k] && a[k] !== '' && a[k] !== 'skipped').length
  const score = Math.round((filled / required.length) * 25)
  return {
    score,
    detail: {
      points: score,
      max: 25,
      filled,
      total: required.length,
      explanation: `${filled} of ${required.length} required ACORD fields completed. ` +
        `Incomplete submissions are the #1 cause of underwriting delays — a complete submission ` +
        `alone materially improves your placement odds.`,
    },
  }
}

// Safety & Compliance — 25 pts
// Source: OSHA — formal safety programs reduce workplace injury claims 40-50%.
// Fire code violations are a hard gate (ACORD Q8).
function scoreSafety(a: Record<string, unknown>): { score: number; detail: Record<string, unknown> } {
  let score = 0
  const items: string[] = []

  // Hard gate: open code violations blocks submission entirely
  if (a['c_code_violations'] === 'yes') {
    return {
      score: 0,
      detail: {
        points: 0, max: 25,
        explanation: '⛔ Open fire/safety code violation detected (ACORD Q8). ' +
          'This blocks submission until corrected. This is not a scoring penalty — ' +
          'it is a hard underwriting gate.',
      },
    }
  }

  // +12 pts: safety manual / OSHA compliance
  // Source: OSHA.gov — formal programs reduce workplace injury claims 40-50%
  if (a['c_safety_manual'] === 'yes') {
    score += 12
    items.push('Has written safety manual (+12 pts, OSHA data: 40–50% fewer injury claims)')
  }

  // +8 pts: no prior cancellations
  if (a['c_prior_cancellation'] === 'no') {
    score += 8
    items.push('No prior policy cancellations (+8 pts)')
  } else if (a['c_prior_cancellation'] === 'yes') {
    items.push('Prior cancellation disclosed (0 pts) — underwriter will review')
  }

  // +5 pts: no discrimination/harassment claims
  if (a['c_discrimination_claims'] === 'no') {
    score += 5
    items.push('No discrimination/harassment claims (+5 pts)')
  }

  return {
    score,
    detail: {
      points: score, max: 25,
      items,
      explanation: items.join('. ') || 'Safety and compliance items not yet answered.',
    },
  }
}

// Physical Property Condition — 20 pts
// Source: IBHS — monitored alarms reduce theft/burglary 60-80%.
// NFPA — sprinklers reduce property loss per fire 50-75%.
function scoreProperty(a: Record<string, unknown>): { score: number; detail: Record<string, unknown> } {
  let score = 0
  const items: string[] = []

  // +6 pts: monitored alarm present
  // Source: IBHS — monitored alarms reduce theft/burglary claim frequency 60-80%
  if (a['d_alarm_system'] === 'yes') {
    score += 6
    items.push('Monitored alarm system present (+6 pts, IBHS: 60–80% fewer theft/burglary claims)')
  }

  // +3 pts: alarm contract uploaded
  if (a['d_alarm_contract'] && a['d_alarm_contract'] !== 'skipped') {
    score += 3
    items.push('Alarm monitoring contract uploaded (+3 pts)')
  }

  // +4 pts: fire sprinklers present
  // Source: NFPA — sprinklers reduce property loss per fire 50-75%
  if (a['d_sprinklers'] === 'yes') {
    score += 4
    items.push('Fire sprinkler system present (+4 pts, NFPA: 50–75% less property loss per fire)')
  }

  // +2 pts: sprinkler certificate uploaded
  if (a['d_sprinkler_certificate'] && a['d_sprinkler_certificate'] !== 'skipped') {
    score += 2
    items.push('Sprinkler inspection certificate uploaded (+2 pts)')
  }

  // +3 pts: fire extinguishers inspected within 12 months
  if (a['d_fire_extinguishers'] === 'yes') {
    score += 3
    items.push('Fire extinguishers present and inspected (+3 pts)')
  }

  // +2 pts: electrical panel photo uploaded (confirms modern panel)
  if (a['d_electrical_photo'] && a['d_electrical_photo'] !== 'skipped') {
    score += 2
    items.push('Electrical panel photo provided (+2 pts)')
  }

  return {
    score,
    detail: { points: score, max: 20, items, explanation: items.join('. ') || 'Physical property items not yet answered.' },
  }
}

// Claims & Financial History — 20 pts
// Source: Prior cancellations (underwriting cause) are near-automatic decline in surplus lines market.
// Clean 5-year loss history is the primary positive signal.
function scoreClaims(a: Record<string, unknown>): { score: number; detail: Record<string, unknown> } {
  let score = 0
  const items: string[] = []

  // +10 pts: no claims in past 5 years
  if (a['e_claims'] === 'no') {
    score += 10
    items.push('No claims in past 5 years (+10 pts)')
  } else if (a['e_claims'] === 'yes') {
    score += 3 // partial — disclosed, brokers can work with disclosed claims
    items.push('Claims disclosed (+3 pts) — disclosed claims are workable; undisclosed ones are not')
  }

  // +5 pts: no bankruptcy
  if (a['c_bankruptcy'] === 'no') {
    score += 5
    items.push('No bankruptcy/foreclosure in past 5 years (+5 pts)')
  }

  // +5 pts: prior carrier info complete (shows insurable history)
  const priorInfoComplete = a['e_prior_carrier'] && a['e_prior_policy_number']
  if (priorInfoComplete) {
    score += 5
    items.push('Prior insurance carrier information provided (+5 pts)')
  }

  return {
    score,
    detail: { points: score, max: 20, items, explanation: items.join('. ') || 'Claims history not yet answered.' },
  }
}

// Neighborhood Context — 10 pts
// Phase 1: Fixed at 5/10 — no SF public data yet (Phase 2 adds live signals)
// Source: Weighted lowest intentionally — public data penalizes businesses for things outside their control.
function scoreNeighborhood(): { score: number; detail: Record<string, unknown> } {
  return {
    score: 5,
    detail: {
      points: 5, max: 10,
      explanation: 'Neighborhood context score is 5/10 in Phase 1. ' +
        'SF public data signals (fire incidents, code violations, flood zone) will be integrated in Phase 2. ' +
        'This dimension is intentionally weighted lowest — public data penalizes businesses for factors outside their control.',
    },
  }
}

export function calculateScore(intakeData: IntakeData): ScoreResult {
  const a = answers(intakeData)

  const doc = scoreDocumentation(a)
  const safety = scoreSafety(a)
  const property = scoreProperty(a)
  const claims = scoreClaims(a)
  const neighborhood = scoreNeighborhood()

  const total = doc.score + safety.score + property.score + claims.score + neighborhood.score

  return {
    total_score: total,
    documentation_score: doc.score,
    safety_score: safety.score,
    property_score: property.score,
    claims_score: claims.score,
    neighborhood_score: neighborhood.score,
    score_details: {
      documentation: doc.detail,
      safety: safety.detail,
      property: property.detail,
      claims: claims.detail,
      neighborhood: neighborhood.detail,
    },
  }
}

export function getScoreTier(score: number): {
  label: string; color: 'green' | 'yellow' | 'orange' | 'red'; meaning: string
} {
  if (score >= 80) return { label: 'Submission Ready', color: 'green', meaning: 'Clean profile. Route to broker review.' }
  if (score >= 60) return { label: 'Minor Gaps', color: 'yellow', meaning: 'Proceed with broker review. Checklist provided.' }
  if (score >= 40) return { label: 'Significant Gaps', color: 'orange', meaning: 'Address gaps before submission.' }
  return { label: 'Not Ready', color: 'red', meaning: 'Advisory mode only. Fix critical items first.' }
}

import { createClient } from '@/lib/supabase/server'
import { NextRequest } from 'next/server'
import { randomBytes } from 'crypto'

export async function POST(req: NextRequest) {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return new Response('Unauthorized', { status: 401 })

  const { business_name, owner_name, owner_contact } = await req.json()

  if (!business_name || !owner_name) {
    return new Response('Missing required fields', { status: 400 })
  }

  const intake_token = randomBytes(16).toString('hex')

  // Ensure broker profile exists
  await supabase.from('brokers').upsert({
    id: user.id,
    name: user.email!.split('@')[0],
    email: user.email!,
  })

  const { data: client, error } = await supabase
    .from('clients')
    .insert({ broker_id: user.id, business_name, owner_name, owner_contact, intake_token })
    .select()
    .single()

  if (error) return new Response(error.message, { status: 500 })

  // Create empty intake_data row
  await supabase.from('intake_data').insert({ client_id: client.id })

  return Response.json(client)
}

import { createClient } from '@/lib/supabase/server'
import { renderToBuffer } from '@react-pdf/renderer'
import { DossierPDF } from '@/lib/pdf/dossier-pdf'
import { createElement } from 'react'
import { notFound } from 'next/navigation'
import { NextRequest } from 'next/server'

export async function GET(
  req: NextRequest,
  { params }: { params: { clientId: string } }
) {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return new Response('Unauthorized', { status: 401 })

  const [{ data: client }, { data: intake }, { data: uploads }, { data: score }] = await Promise.all([
    supabase.from('clients').select('*').eq('id', params.clientId).eq('broker_id', user.id).single(),
    supabase.from('intake_data').select('*').eq('client_id', params.clientId).single(),
    supabase.from('uploads').select('*').eq('client_id', params.clientId).order('uploaded_at'),
    supabase.from('readiness_scores').select('*').eq('client_id', params.clientId).single(),
  ])

  if (!client) return new Response('Not found', { status: 404 })

  const buffer = await renderToBuffer(
    createElement(DossierPDF, {
      client,
      intake: intake ?? null,
      uploads: uploads ?? [],
      score: score ?? null,
      generatedAt: new Date().toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' }),
    })
  )

  return new Response(buffer, {
    headers: {
      'Content-Type': 'application/pdf',
      'Content-Disposition': `attachment; filename="${client.business_name.replace(/\s+/g, '_')}_dossier.pdf"`,
    },
  })
}
